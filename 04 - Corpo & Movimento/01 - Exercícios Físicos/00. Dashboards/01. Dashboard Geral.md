## [[Guia Dashboards Exercícios#🧬 Snapshot de Variação Corporal|🧬 Snapshot de Variação Corporal]]

> [!multi-column]
>
>> [!info] *Comparativo entre a primeira e a última medição registrada nos últimos 90 dias.*
>>
>> ```dataviewjs
>> // --- CONFIGURAÇÃO ---
>> const diasParaAnalise = 90;
>> const logPath = "99 - BACKEND/Logs_Metricas/Exercicios/body_measures_log.md";
>> const moment = window.moment;
>> 
>> // --- FUNÇÕES AUXILIARES ---
>> const createCard = (titulo, valor, unidade, cor) => {
>>     let valorFormatado = valor;
>>     let sinal = '';
>>     if (typeof valor === 'number') {
>>         valorFormatado = valor.toFixed(1);
>>         if (valor > 0) sinal = '+';
>>     }
>> 
>>     return `
>>     <div style="flex: 1; min-width: 150px; background-color: var(--background-secondary); border: 1px solid var(--background-modifier-border); border-left: 4px solid ${cor}; border-radius: 6px; padding: 15px; text-align: center;">
>>         <div style="font-size: 0.9em; color: var(--text-muted); font-weight: 600; margin-bottom: 8px;">${titulo}</div>
>>         <div style="font-size: 1.8em; font-weight: bold; color: var(--text-normal);">${sinal}${valorFormatado} <span style="font-size: 0.6em; color: var(--text-faint);">${unidade}</span></div>
>>     </div>
>>     `;
>> };
>> 
>> // --- LÓGICA PRINCIPAL ---
>> const logFile = app.vault.getAbstractFileByPath(logPath);
>> if (!logFile) {
>>     dv.paragraph("ℹ️ Nenhum log de medidas encontrado.");
>> } else {
>>     const content = await app.vault.cachedRead(logFile);
>>     const logs = content.split('\n')
>>         .filter(line => line.trim().startsWith('- (data::'))
>>         .map(line => {
>>             const data = line.match(/\(data::\s*([^)]+)\)/)?.[1];
>>             const peso = parseFloat(line.match(/\(peso_kg::\s*([\d.]+)\)/)?.[1]);
>>             const cintura = parseFloat(line.match(/\(cintura_cm::\s*([\d.]+)\)/)?.[1]);
>>             const peitoral = parseFloat(line.match(/\(peitoral_cm::\s*([\d.]+)\)/)?.[1]);
>>             return { data, peso, cintura, peitoral };
>>         })
>>         .filter(log => log.data && moment(log.data).isAfter(moment().subtract(diasParaAnalise, 'days')))
>>         .sort((a, b) => a.data.localeCompare(b.data));
>> 
>>     if (logs.length < 2) {
>>         dv.paragraph("ℹ️ São necessários pelo menos dois registros nos últimos " + diasParaAnalise + " dias para calcular a variação.");
>>     } else {
>>         const primeiro = logs[0];
>>         const ultimo = logs[logs.length - 1];
>> 
>>         const variacaoPeso = ultimo.peso && primeiro.peso ? ultimo.peso - primeiro.peso : "N/A";
>>         const variacaoCintura = ultimo.cintura && primeiro.cintura ? ultimo.cintura - primeiro.cintura : "N/A";
>>         const variacaoPeitoral = ultimo.peitoral && primeiro.peitoral ? ultimo.peitoral - primeiro.peitoral : "N/A";
>> 
>>         let html = `<div style="display: flex; gap: 15px; justify-content: space-around; flex-wrap: wrap;">`;
>>         html += createCard("Variação de Peso", variacaoPeso, "kg", "var(--color-red-rgb)");
>>         html += createCard("Variação de Cintura", variacaoCintura, "cm", "var(--color-yellow-rgb)");
>>         html += createCard("Variação de Peitoral", variacaoPeitoral, "cm", "var(--color-blue-rgb)");
>>         html += `</div>`;
>>         
>>         const diasDesdeUltima = moment().diff(moment(ultimo.data), 'days');
>>         const textoUltima = `Última medição registrada há ${diasDesdeUltima} dia(s), em ${moment(ultimo.data).format("DD/MM/YYYY")}.`;
>>         dv.paragraph(`<div style="font-size: 0.8em; color: var(--text-faint); text-align: center; margin-top: 10px;">${textoUltima}</div>`);
>>         
>>         dv.paragraph(html);
>>     }
>> }
>> ```

## [[Guia Dashboards Exercícios#🧬 Tendências Corporais de Longo Prazo|🧬 Tendências Corporais de Longo Prazo]]

> [!multi-column]
>
>> [!info] *Análise da evolução das principais métricas corporais ao longo do tempo. Use este gráfico para uma visão macro da sua jornada física.*
>>
>> ```dataviewjs
>> const logPath = "99 - BACKEND/Logs_Metricas/Exercicios/body_measures_log.md";
>> const logFile = app.vault.getAbstractFileByPath(logPath);
>> 
>> if (!logFile) {
>>     dv.paragraph("ℹ️ Nenhum log de medidas corporais encontrado.");
>> } else {
>>     const content = await app.vault.cachedRead(logFile);
>>     const logs = content.split('\n').filter(line => line.trim().startsWith('- (data::'));
>> 
>>     if (logs.length < 2) { // Precisa de pelo menos 2 pontos para uma linha
>>         dv.paragraph("ℹ️ Registros insuficientes para gerar um gráfico de tendências (mínimo 2).");
>>     } else {
>>         const chartData = [];
>> 
>>         logs.forEach(log => {
>>             const data = log.match(/\(data::\s*([^)]+)\)/)?.[1];
>>             if (!data) return;
>> 
>>             // Extrai cada métrica, tratando valores ausentes como 'null' para o gráfico
>>             const peso = parseFloat(log.match(/\(peso_kg::\s*([\d.]+)\)/)?.[1]) || null;
>>             const cintura = parseFloat(log.match(/\(cintura_cm::\s*([\d.]+)\)/)?.[1]) || null;
>>             const peitoral = parseFloat(log.match(/\(peitoral_cm::\s*([\d.]+)\)/)?.[1]) || null;
>> 
>>             if (peso) chartData.push({ data, metrica: "Peso (kg)", valor: peso });
>>             if (cintura) chartData.push({ data, metrica: "Cintura (cm)", valor: cintura });
>>             if (peitoral) chartData.push({ data, metrica: "Peitoral (cm)", valor: peitoral });
>>         });
>> 
>>         // Ordena por data para garantir que a linha do tempo esteja correta
>>         chartData.sort((a, b) => a.data.localeCompare(b.data));
>> 
>>         // Usa o plugin ChartsView que já está no seu sistema
>>         const yamlString = `
>> \`\`\`chartsview
>> type: Line
>> data: ${JSON.stringify(chartData)}
>> options:
>>   xField: 'data'
>>   yField: 'valor'
>>   seriesField: 'metrica'
>>   smooth: true
>>   yAxis:
>>     title: { text: 'Valor (kg/cm)', style: { fill: '#cccccc' } }
>>   xAxis: 
>>     title: { text: 'Data', style: { fill: '#cccccc' } }
>>     type: 'time'
>>     label: { autoRotate: true, autoHide: false }
>>   color: ['#e74c3c', '#f1c40f', '#3498db']
>>   legend: { position: 'top-left' }
>>   point:
>>     shape: 'circle'
>>     size: 4
>>   tooltip:
>>     formatter: |
>>       function(datum) {
>>         const unidade = datum.metrica.includes('(kg)') ? 'kg' : 'cm';
>>         return { name: datum.metrica, value: datum.valor.toFixed(1) + ' ' + unidade };
>>       }
>>   title: { visible: true, text: 'Evolução da Composição Corporal' }
>> \`\`\`
>> `;
>>         dv.paragraph(yamlString);
>>     }
>> }
>> ```

>> type: Line

>>   dataviewjs:
>>   if (!dv) { return []; }
>>
>>   const nomeExercicioSelecionado = dv.current().exercicio_selecionado;
>>   if (!nomeExercicioSelecionado || nomeExercicioSelecionado.trim() === "") {
>>     return [];
>>   }
>>   
>>   const sanitizarParaId = (str) => {
>>       if (!str) return "";
>>       return str.normalize("NFD").replace(/[\u00c0-\u024f]/g, "").toLowerCase().trim().replace(/[-\s]+/g, '_').replace(/[^\w_]+/g, '');
>>   };
>>
>>   const exercicioId = sanitizarParaId(nomeExercicioSelecionado);
>>
>>   // --- LÓGICA DE BUSCA HIERÁRQUICA ---
>>   const logsRootPath = "99 - BACKEND/Logs_Metricas/Exercicios";
>>   
>>   const encontrarArquivoLog = (id) => {
>>       const root = app.vault.getAbstractFileByPath(logsRootPath);
>>       if (!root || !root.children) return null;
>>   
>>       for (const grupoFolder of root.children) {
>>           if (!grupoFolder.extension) { // É pasta
>>               const path = `${grupoFolder.path}/${id}/raw_logs.md`;
>>               const file = app.vault.getAbstractFileByPath(path);
>>               if (file) return file;
>>           }
>>       }
>>       return null;
>>   };
>>
>>   const logFile = encontrarArquivoLog(exercicioId);
>>   if (!logFile) { return []; }
>>
>>   // --- RESTO DO CÓDIGO PERMANECE O MESMO ---
>>   const content = await app.vault.cachedRead(logFile);
>>   const logsDoExercicio = content.split('\n').filter(line => line.includes("(sessao_fim::WORK)"));
>>
>>   if (logsDoExercicio.length === 0) { return []; }
>>
>>   const seriesPorDia = logsDoExercicio.reduce((acc, log) => {
>>       const data = log.match(/\(log_date::\s*([^)]+)\)/)?.[1];
>>       const carga = parseFloat(log.match(/\(carga_kg::\s*([\d.]+)\)/)?.[1] || '0');
>>       const reps = parseInt(log.match(/\(reps_feito::\s*(\d+)\)/)?.[1] || '0');
>>       
>>       if (data && carga > 0 && reps > 0) {
>>           if (!acc[data] || carga > acc[data].carga) {
>>               acc[data] = { carga, reps };
>>           }
>>       }
>>       return acc;
>>   }, {});
>>
>>   const brzycki1RM = (carga, reps) => {
>>       if (reps > 12) return null;
>>       return carga / (1.0278 - (0.0278 * reps));
>>   };
>>   
>>   const chartData = Object.entries(seriesPorDia).flatMap(([dia, { carga, reps }]) => {
>>       const rmEstimado = brzycki1RM(carga, reps);
>>       const dataPoints = [{ dia, tipo: 'Carga Máxima', valor: carga }];
>>       if (rmEstimado !== null) {
>>           dataPoints.push({ dia, tipo: '1RM Estimado', valor: Math.round(rmEstimado) });
>>       }
>>       return dataPoints;
>>   });
>>   
>>   return chartData.sort((a, b) => a.dia.localeCompare(b.dia));
>> options:
>>   xField: 'dia'
>>   yField: 'valor'
>>   seriesField: 'tipo'
>>   smooth: true
>>   yAxis: { title: { text: 'Peso (kg)', style: { fill: '#cccccc' } } }
>>   xAxis: { label: { autoRotate: true, autoHide: false }, title: { text: 'Data', style: { fill: '#cccccc' } } }
>>   legend: { position: 'top' }
>>   color: ['#3498db', '#e74c3c']
>>   tooltip:
>>     formatter: |
>>       function(datum) {
>>         return { name: datum.tipo, value: datum.valor + ' kg' };
>>       }
>>   point: { shape: 'circle', size: 4 }
>>   title: { visible: true, text: 'Progressão de Carga & 1RM Estimado' }

## [[Guia Dashboards Exercícios#📔 Diário do Atleta (Qualitativo)|📔 Diário do Atleta (Qualitativo)]]

> [!tip] Inteligência Subjetiva
> Este painel filtra automaticamente sessões com **notas manuais**, **RPE extremo (9-10)** ou **má qualidade técnica (<5)**. Use para detectar padrões de lesão ou overtraining.

```dataviewjs
await dv.view("99 - BACKEND/Scripts/Views/Exercicios/renderDiarioAtleta", {});
```
