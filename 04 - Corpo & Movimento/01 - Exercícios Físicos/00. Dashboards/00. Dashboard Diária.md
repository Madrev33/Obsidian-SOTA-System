# Dashboard Diário de Exercícios

> [!summary] Visão Tática
> Este dashboard é seu companheiro de treino diário. Ele oferece um resumo em tempo real do esforço do dia, detalhando cada exercício e cada série à medida que são registrados.

---

```dataviewjs
// CONFIGURAÇÃO DOS CAMINHOS
const sessaoFolder = "04 - Corpo & Movimento/01 - Exercícios Físicos/03. Sessões de Treino";
const manualFolder = "04 - Corpo & Movimento/01 - Exercícios Físicos/01. Manual/Exercícios";

// 1. IDENTIFICAR A SESSÃO DE HOJE
const hoje = moment();
const pathSessaoHoje = `${sessaoFolder}/${hoje.format("YYYY")}/${hoje.format("MM")}/${hoje.format("YYYY-MM-DD")}.md`;
const pageSessao = dv.page(pathSessaoHoje);

if (!pageSessao) {
    dv.paragraph(`> [!warning] **Sem Agendamento**\n> Nenhuma nota de sessão encontrada para hoje (${hoje.format("DD/MM")}).`);
} else {
    // 2. EXTRAIR IDS DOS EXERCÍCIOS DAS TAGS DAS TAREFAS
    const idsExerciciosHoje = new Set();
    
    if (pageSessao.file.tasks) {
        pageSessao.file.tasks.forEach(t => {
            t.tags.forEach(tag => {
                // 👇 ALTERAÇÃO AQUI: Regex para capturar a terceira parte (ID) 👇
                // #exercicio/{grupo}/{id}
                const match = tag.match(/#exercicio\/[^\/]+\/([^\/\s]+)/);
                if (match && match[1]) {
                    idsExerciciosHoje.add(match[1]); // Adiciona o ID (ex: supino_reto) ao Set
                }
            });
        });
    }

    if (idsExerciciosHoje.size === 0) {
        dv.paragraph(`> [!info] Sessão criada, mas sem exercícios com tags detectados ainda.`);
        dv.paragraph(`[[${pathSessaoHoje}|📝 Abrir Nota da Sessão]]`);
    } else {
        // 3. BUSCAR AS NOTAS DO MANUAL QUE CORRESPONDEM AOS IDS
        const exerciciosDoDia = dv.pages(`"${manualFolder}"`)
            .where(p => idsExerciciosHoje.has(p.exercicio_id)) // Agora a correspondência vai funcionar
            .sort(p => p.file.name);

        // 4. RENDERIZAR TABELA DE NAVEGAÇÃO
        dv.header(3, "🚀 Atalhos para o Treino de Hoje");
        
        dv.table(
            ["Exercício (Ir para Manual)", "Grupo", "Equipamento"],
            exerciciosDoDia.map(p => [
                p.file.link, 
                // Formata o ID do grupo para exibição (ex: peitoral -> Peitoral)
                p.grupo_muscular_primario ? p.grupo_muscular_primario.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase()) : "-",
                p.equipamento || "-"
            ])
        );

        // 5. BOTÃO CORRIGIDO
        dv.el("br", ""); // Quebra de linha
        dv.el("button", "📝 Abrir Nota de Treino", { 
            attr: { onclick: `app.workspace.openLinkText('${pathSessaoHoje}', '', false)` },
            cls: "sota-button cta" // Usa seu estilo CSS existente
        });
    }
}
```
