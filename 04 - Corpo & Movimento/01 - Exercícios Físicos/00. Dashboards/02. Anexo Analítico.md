## [[Guia Dashboards Exercícios#📏 Histórico de Medidas Corporais|📏 Histórico de Medidas Corporais]]

> [!multi-column]
>
>> [!info] *Todos os registros de medidas corporais coletados ao longo do tempo. Use para identificar tendências e correlações com seu treinamento.*
>>
>> ```dataviewjs
>> const logPath = "99 - BACKEND/Logs_Metricas/Exercicios/body_measures_log.md";
>> const logFile = app.vault.getAbstractFileByPath(logPath);
>> 
>> if (!logFile) {
>>     dv.paragraph("ℹ️ Nenhum log de medidas corporais encontrado.");
>> } else {
>>     const content = await app.vault.cachedRead(logFile);
>>     const logs = content.split('\n').filter(line => line.trim().startsWith('- (data::'));
>> 
>>     if (logs.length === 0) {
>>         dv.paragraph("Nenhum registro encontrado no arquivo de log.");
>>     } else {
>>         const headers = [
>>             "Data", "Peso (kg)", "IMC", "RCQ", "Pescoço", "Peitoral", "Cintura", "Quadril", 
>>             "Braço D", "Braço E", "Coxa D", "Coxa E", "Pant. D", "Pant. E"
>>         ];
>> 
>>         // Objeto para acumular totais para o cálculo da média
>>         const totals = { count: 0 };
>>         headers.slice(1, -1).forEach(h => totals[h] = 0); // Inicializa todos, menos Data e Fotos
>> 
>>         const rows = logs.map(log => {
>>             const getData = (key) => log.match(new RegExp(`\\(${key}::\\s*([^)]+)\\)`))?.[1] || "-";
>>             
>>             const rowData = {
>>                 "Data": moment(getData("data")).format("DD/MM/YYYY"),
>>                 "Peso (kg)": parseFloat(getData("peso_kg")) || null,
>>                 "IMC": parseFloat(getData("imc_calculado")) || null,
>>                 "RCQ": parseFloat(getData("rcq_calculado")) || null,
>>                 "Pescoço": parseFloat(getData("pescoco_cm")) || null,
>>                 "Peitoral": parseFloat(getData("peitoral_cm")) || null,
>>                 "Cintura": parseFloat(getData("cintura_cm")) || null,
>>                 "Quadril": parseFloat(getData("quadril_cm")) || null,
>>                 "Braço D": parseFloat(getData("braco_d_cm")) || null,
>>                 "Braço E": parseFloat(getData("braco_e_cm")) || null,
>>                 "Coxa D": parseFloat(getData("coxa_d_cm")) || null,
>>                 "Coxa E": parseFloat(getData("coxa_e_cm")) || null,
>>                 "Pant. D": parseFloat(getData("panturrilha_d_cm")) || null,
>>                 "Pant. E": parseFloat(getData("panturrilha_e_cm")) || null,
>>             };
>> 
>>             // Acumula os totais se o valor for numérico
>>             if (rowData["Peso (kg)"]) {
>>                 totals.count++;
>>                 Object.keys(rowData).forEach(key => {
>>                     if (typeof rowData[key] === 'number') {
>>                         totals[key] = (totals[key] || 0) + rowData[key];
>>                     }
>>                 });
>>             }
>> 
>>             // Retorna os valores para a linha da tabela, substituindo null por "-"
>>             return headers.map(h => rowData[h] === null ? "-" : rowData[h]);
>>         });
>> 
>>         // Calcula a linha de Média
>>         const averageRow = headers.map(h => {
>>             if (h === "Data") return `**Média (${totals.count})**`;
>>             
>>             const average = totals.count > 0 ? (totals[h] / totals.count) : 0;
>>             return average > 0 ? `**${average.toFixed(1)}**` : "-";
>>         });
>> 
>>         rows.push(averageRow);
>> 
>>         dv.table(headers, rows);
>>     }
>> }
>> ```

## [[Guia Dashboards Exercícios#🏆 KPIs de Eficiência de Força|🏆 KPIs de Eficiência de Força]]

> [!multi-column]
>
>> [!info] *Análise de força normalizada pelo peso corporal. A Força Relativa indica a eficiência do seu levantamento.*
>>
>> ```dataviewjs
>> // --- CONFIGURAÇÃO ---
>> const config = {
>>     corpo_logs_path: "99 - BACKEND/Logs_Metricas/Exercicios/body_measures_log.md",
>>     exercicios_chave: ["agachamento_livre", "supino_reto_com_barra", "levantamento_terra"],
>>     logs_root: "99 - BACKEND/Logs_Metricas/Exercicios"
>> };
>> 
>> // --- FUNÇÕES DE CÁLCULO (Validado: Brzycki) ---
>> const brzycki1RM = (carga, reps) => {
>>     // Trava estatística: Acima de 12 reps a estimativa perde precisão
>>     if (reps > 12) return null; 
>>     return carga / (1.0278 - (0.0278 * reps));
>> };
>> 
>> // --- FUNÇÃO DE BUSCA DE ARQUIVO (Hierarchical Sharding Support) ---
>> // Procura o arquivo de log do exercício dentro de qualquer subpasta de grupo muscular
>> const encontrarArquivoLog = (exercicioId) => {
>>     const root = app.vault.getAbstractFileByPath(config.logs_root);
>>     if (!root || !root.children) return null;
>> 
>>     // 1. Tenta caminho legado (Raiz)
>>     let path = `${config.logs_root}/${exercicioId}/raw_logs.md`;
>>     let file = app.vault.getAbstractFileByPath(path);
>>     if (file) return file;
>> 
>>     // 2. Varre subpastas (Grupos)
>>     for (const grupoFolder of root.children) {
>>         if (!grupoFolder.extension) { // Garante que é pasta
>>             path = `${grupoFolder.path}/${exercicioId}/raw_logs.md`;
>>             file = app.vault.getAbstractFileByPath(path);
>>             if (file) return file;
>>         }
>>     }
>>     return null;
>> };
>> 
>> // --- PROCESSAMENTO DE DADOS ---
>> const corpoLogFile = app.vault.getAbstractFileByPath(config.corpo_logs_path);
>> 
>> if (!corpoLogFile) {
>>     dv.paragraph("⚠️ Arquivo de log de medidas corporais não encontrado.");
>> } else {
>>     const corpoContent = await app.vault.cachedRead(corpoLogFile);
>> 
>>     const logsCorpo = corpoContent.split('\n').map(line => {
>>         const dataMatch = line.match(/\(data::\s*([^)]+)\)/);
>>         const pesoMatch = line.match(/\(peso_kg::\s*([\d.]+)\)/);
>>         return (dataMatch && pesoMatch) ? { data: dataMatch[1], peso: parseFloat(pesoMatch[1]) } : null;
>>     }).filter(Boolean);
>> 
>>     const encontrarPesoMaisProximo = (dataAlvo) => {
>>         if (logsCorpo.length === 0) return null;
>>         const momentoAlvo = moment(dataAlvo);
>>         let pesoMaisProximo = logsCorpo[0].peso;
>>         let menorDiferenca = Math.abs(momentoAlvo.diff(moment(logsCorpo[0].data), 'days'));
>>         
>>         for (let i = 1; i < logsCorpo.length; i++) {
>>             const diferenca = Math.abs(momentoAlvo.diff(moment(logsCorpo[i].data), 'days'));
>>             if (diferenca < menorDiferenca) {
>>                 menorDiferenca = diferenca;
>>                 pesoMaisProximo = logsCorpo[i].peso;
>>             }
>>         }
>>         return pesoMaisProximo;
>>     };
>> 
>>     let records = [];
>> 
>>     // Itera sobre os exercícios chave
>>     for (const id of config.exercicios_chave) {
>>         const logFile = encontrarArquivoLog(id);
>>         
>>         if (logFile) {
>>             const content = await app.vault.cachedRead(logFile);
>>             const logsExercicio = content.split('\n').filter(line => line.includes("sessao_fim::WORK"));
>> 
>>             let recordeExercicio = { rm: 0, data: null, carga: 0, reps: 0 };
>>             
>>             logsExercicio.forEach(log => {
>>                 const carga = parseFloat(log.match(/\(carga_kg::\s*([\d.]+)\)/)?.[1] || '0');
>>                 const reps = parseInt(log.match(/\(reps_feito::\s*(\d+)\)/)?.[1] || '0');
>>                 const rm = brzycki1RM(carga, reps);
>>                 
>>                 // Atualiza se o RM estimado for maior (Melhor performance absoluta)
>>                 if (rm && rm > recordeExercicio.rm) {
>>                     recordeExercicio = { 
>>                         rm: rm, 
>>                         data: log.match(/\(log_date::\s*([^)]+)\)/)?.[1], 
>>                         carga: carga, 
>>                         reps: reps 
>>                     };
>>                 }
>>             });
>> 
>>             if (recordeExercicio.data) {
>>                 const pesoCorporal = encontrarPesoMaisProximo(recordeExercicio.data);
>>                 const forcaRelativa = pesoCorporal ? (recordeExercicio.rm / pesoCorporal).toFixed(2) : "N/A";
>>                 
>>                 records.push([
>>                     id.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase()),
>>                     `**${Math.round(recordeExercicio.rm)} kg** (${recordeExercicio.carga}kg x ${recordeExercicio.reps})`,
>>                     moment(recordeExercicio.data).format("DD/MM/YYYY"),
>>                     pesoCorporal ? `${pesoCorporal.toFixed(1)} kg` : "N/A",
>>                     forcaRelativa ? `${forcaRelativa}x` : "N/A"
>>                 ]);
>>             }
>>         }
>>     }
>> 
>>     if (records.length > 0) {
>>         const headers = ["Exercício", "1RM Estimado (Carga x Reps)", "Data do RP", "Peso Corporal (Aprox.)", "Força Relativa"];
>>         dv.table(headers, records);
>>     } else {
>>         dv.paragraph("Nenhum recorde encontrado para os exercícios chave.");
>>     }
>> }
>> ```
